Notification emails alert users about activity in your application.

## Variants

### Default

Simple notification with actor info, action, and CTA button.

```tsx title="registry/bases/mjml-react/blocks/notification-default.tsx"
import {
  Mjml,
  MjmlAll,
  MjmlAttributes,
  MjmlBody,
  MjmlColumn,
  MjmlHead,
  MjmlPreview,
  MjmlSection,
  MjmlText,
  MjmlWrapper,
} from "@faire/mjml-react";

import { resolveEmailTheme } from "@/registry/lib/resolve-theme";
import { defaultTheme } from "@/registry/themes/default";
import { emailMjmlLayoutVars } from "../email-mjml-layout";

type Props = Record<string, never>;

export const NotificationDefault = (_props: Props) => {
  const theme = defaultTheme;
  const t = resolveEmailTheme(theme);
  const { baseFs, bg, fg, lh, sans, width } = emailMjmlLayoutVars(t);

  return (
    <Mjml>
      <MjmlHead>
        <MjmlPreview>Email preview</MjmlPreview>
        <MjmlAttributes>
          <MjmlAll color={fg} fontFamily={sans} />
          <MjmlText fontSize={baseFs} lineHeight={lh} />
        </MjmlAttributes>
      </MjmlHead>
      <MjmlBody backgroundColor={bg} width={width}>
        <MjmlWrapper padding="0">
          <MjmlSection padding={`${t.spacing.xl ?? "24px"} 0`}>
            <MjmlColumn>
              <MjmlText
                color={t.colors.foreground}
                fontFamily={t.fontFamily.sans}
              >
                {/* … */}
              </MjmlText>
            </MjmlColumn>
          </MjmlSection>
        </MjmlWrapper>
      </MjmlBody>
    </Mjml>
  );
};
```


## Props

| Prop             | Type     | Default          | Description                               |
| ---------------- | -------- | ---------------- | ----------------------------------------- |
| `logoUrl`        | `string` | -                | URL of the product logo                   |
| `actorName`      | `string` | `"Someone"`      | Name of the user who performed the action |
| `actorAvatarUrl` | `string` | -                | URL of the user's avatar                  |
| `_action`        | `string` | `"notified you"` | Action description                        |
| `_targetName`    | `string` | `"your update"`  | Name of the notification target           |
| `ctaLabel`       | `string` | `"View"`         | CTA button label                          |
| `ctaHref`        | `string` | `"#"`            | CTA button link                           |
| `_productName`   | `string` | `"Acme"`         | Product name                              |
