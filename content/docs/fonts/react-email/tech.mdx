## Usage

```tsx
import { TechFonts } from "@/components/email/tech-fonts";

export default function Email() {
  return (
    <Html>
      <Head>
        <TechFonts />
      </Head>
      <Body>{/* email content */}</Body>
    </Html>
  );
}
```

## Font Details

| Property       | Value                                                                                                             |
| -------------- | ----------------------------------------------------------------------------------------------------------------- |
| Primary Font   | Geist                                                                                                             |
| Secondary Font | Inter                                                                                                             |
| Fallback       | Arial, sans-serif                                                                                                 |
| Weights        | 400, 450, 500, 600, 700                                                                                           |
| Source         | [Resend Studio demo](https://github.com/resend/react-email/blob/canary/apps/demo/emails/05-Studio/tech-fonts.tsx) |
