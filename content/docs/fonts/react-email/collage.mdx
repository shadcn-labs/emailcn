## Usage

```tsx
import { CollageFonts } from "@/components/email/collage-fonts";

export default function Email() {
  return (
    <Html>
      <Head>
        <CollageFonts />
      </Head>
      <Body>{/* email content */}</Body>
    </Html>
  );
}
```
