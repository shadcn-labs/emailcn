## Usage

```tsx
import { DefaultFonts } from "@/components/email/default";

export default function Email() {
  return (
    <Html>
      <Head>
        <DefaultFonts />
      </Head>
      <Body>{/* email content */}</Body>
    </Html>
  );
}
```
