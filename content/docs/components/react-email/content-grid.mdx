The Content Grid component displays content in a responsive grid layout.

## Usage

```tsx
import {
  ContentGrid,
  ContentGridItem,
} from "@/registry/bases/react-email/ui/content-grid";

export default function Email() {
  return (
    <ContentGrid>
      <ContentGridItem>Item 1</ContentGridItem>
      <ContentGridItem>Item 2</ContentGridItem>
    </ContentGrid>
  );
}
```

## Props

| Prop      | Type         | Default        | Description         |
| --------- | ------------ | -------------- | ------------------- |
| `theme`   | `EmailTheme` | `defaultTheme` | Email theme styling |
| `columns` | `number`     | `2`            | Number of columns   |
| `gap`     | `string`     | -              | Gap between items   |
