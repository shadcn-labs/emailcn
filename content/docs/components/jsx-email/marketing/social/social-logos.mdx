## Usage

```tsx
import { SocialLogos } from "@/components/emails/marketing/social/social-logos";
```

```tsx
<SocialLogos {...SocialLogos.PreviewProps} tile="circle" />
<SocialLogos {...SocialLogos.PreviewProps} tile="outlined-pill" variant="slanted-left" />
```

## API Reference

| Prop      | Type                                                                                                  | Default     | Description                |
| --------- | ----------------------------------------------------------------------------------------------------- | ----------- | -------------------------- |
| `tile`    | `"square" \| "circle" \| "pill" \| "squared-box" \| "outlined-box" \| "outlined-pill" \| "outlined-square"` | `"square"`  | Tile shape around each icon |
| `variant` | `"default" \| "slanted-left" \| "slanted-right"`                                                      | `"default"` | Row tilt                   |
| `theme`   | `EmailThemeTokens`                                                                                      | `defaultTheme` | Email theme styling     |
