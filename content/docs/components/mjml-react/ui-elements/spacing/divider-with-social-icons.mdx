## Usage

```tsx
import { DividerWithSocialIcons } from "@/components/emails/ui-elements/spacing/divider-with-social-icons";
```

```tsx
<DividerWithSocialIcons
  {...DividerWithSocialIcons.PreviewProps}
  variant="left"
/>
```

## API Reference

| Prop      | Type                            | Default        | Description             |
| --------- | ------------------------------- | -------------- | ----------------------- |
| `variant` | `"center" \| "left" \| "right"` | `"center"`     | Inline content position |
| `theme`   | `EmailThemeTokens`              | `defaultTheme` | Email theme styling     |
